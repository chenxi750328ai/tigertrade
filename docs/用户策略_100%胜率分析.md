### 止盈决策逻辑

```python
# 伪代码
def should_take_profit(current_price, avg_cost):
    """判断是否止盈"""
    
    # 1. 必须盈利
    if current_price <= avg_cost:
        return False
    
    # 2. 收集各时间框架信号
    signals = []
    
    # 1分钟信号
    if rsi_1m > 70:
        signals.append('1m_rsi_overbought')
    if volume_1m < avg_volume * 0.8:
        signals.append('1m_volume_decrease')
    
    # 2分钟信号
    if has_long_upper_shadow(kline_2m):
        signals.append('2m_upper_shadow')
    
    # 3分钟信号
    if price_momentum_decreasing(kline_3m):
        signals.append('3m_momentum_down')
    
    # 5分钟信号
    if macd_bearish_divergence(kline_5m):
        signals.append('5m_macd_divergence')
    if price >= boll_5m_upper * 0.95:
        signals.append('5m_near_upper_band')
    
    # 10分钟信号
    if near_resistance_level(kline_10m):
        signals.append('10m_resistance')
    
    # 3. 综合判断
    # 如果有3个以上见顶信号，止盈
    if len(signals) >= 3:
        return True
    
    # 如果5分钟和10分钟都有信号，止盈
    if any('5m' in s for s in signals) and any('10m' in s for s in signals):
        return True
    
    return False
```
